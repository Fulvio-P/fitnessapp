Comando per lanciare l'immagine:
    $ docker run -d --name <nome> -v <dir host>:/var/lib/postgresql/data -p <porta host>:5432 [-e POSTGRES_USER=<username>] -e POSTGRES_PASSWORD=<password> [-e POSTGRES_DB=<nome db>] postgres


sudo docker run -d --name elefante -v /var/lib/postgresql/data:/var/lib/postgresql/data -p 5432:5432 -e POSTGRES_USER=admin -e POSTGRES_PASSWORD=pass -e POSTGRES_DB=prova postgres
